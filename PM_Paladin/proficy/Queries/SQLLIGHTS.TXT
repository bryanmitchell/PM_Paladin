SELECT pl.RemoteIoAddress, ws.GreenLightModuleNumber, ws.GreenLightPointNumber, ws.GreenLightOn, ws.YellowLightModuleNumber, ws.YellowLightPointNumber, ws.YellowLightOn, ws.RedLightModuleNumber, ws.RedLightPointNumber, ws.RedLightOn INTO (>>LineAddress, >>GreenModule, >>GreenPoint, >>GreenOn, >>YellowModule, >>YellowPoint, >>YellowOn, >>RedModule, >>RedPoint, >>RedOn) FROM ProductionLine pl INNER JOIN Workstation ws ON pl.LineID = ws.LineID;
